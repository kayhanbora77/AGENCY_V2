SELECT *
FROM ta_standard_middleeast t
WHERE t.eueligible = TRUE  
  AND t.FromAirport IN (
      SELECT a.CodeIataAirport
      FROM airports a
      WHERE a.CodeIso2Country = 'TR'
  )
  AND t.ToAirport IN (
      SELECT a.CodeIataAirport
      FROM airports a
      WHERE a.CodeIso2Country = 'TR'
  );
