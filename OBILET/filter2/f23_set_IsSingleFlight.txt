UPDATE TA_STANDARD_OBILET
SET IsSingleFlight = TRUE
WHERE ConnectionId IN (
    SELECT ConnectionId
    FROM TA_STANDARD_OBILET
    GROUP BY ConnectionId
    HAVING COUNT(*) = 1
);


UPDATE TA_STANDARD_OBILET
SET IsSingleFlight = FALSE
WHERE ConnectionId IN (
    SELECT ConnectionId
    FROM TA_STANDARD_OBILET
    GROUP BY ConnectionId
    HAVING COUNT(*) > 1
);
