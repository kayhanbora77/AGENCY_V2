UPDATE TA_STANDARD_AKBAR_V2
SET IsSingleFlight = TRUE
WHERE ConnectionId IN (
    SELECT ConnectionId
    FROM TA_STANDARD_AKBAR_V2
    GROUP BY ConnectionId
    HAVING COUNT(*) = 1
);


UPDATE TA_STANDARD_AKBAR_V2
SET IsSingleFlight = FALSE
WHERE ConnectionId IN (
    SELECT ConnectionId
    FROM TA_STANDARD_AKBAR_V2
    GROUP BY ConnectionId
    HAVING COUNT(*) > 1
);
