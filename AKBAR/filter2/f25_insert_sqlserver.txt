    PaxName             VARCHAR(500) NULL,
    AgencyRefNumber     VARCHAR(100) NULL,
    ETicketNo           VARCHAR(100) NULL,

    EUEligibleDuration  INT NULL,
    ExtraNote           VARCHAR(MAX) NULL,

    FlightFound         BIT NULL,
    LegNo               INT NULL,

    IsTimeLimitL1       BIT NULL,
    IsTimeLimitL2       BIT NULL,

    EUFlights_Id        BIGINT NULL,
    Link_Id             VARCHAR(100) NULL,

    DelayInSecond       INT NULL,
    Status              VARCHAR(50) NULL,

    IsSingleFlight      BIT NULL
);

-- Step 4: Insert into real table with conversion
INSERT INTO dbo.TA_STANDARD_AKBAR_V2
SELECT
    TRY_CAST(Id AS UNIQUEIDENTIFIER),
    ConnectionID,
    PaxName,
    AgencyRefNumber,
    ETicketNo,
    FlightNumber,
    TRY_CAST(DepartureDate AS DATE) AS DepartureDate,
    FileName,
    BookingRef,
    AirlineCode,
    FromAirport,
    ToAirport,
    LastLegAirport,
    CAST(EUEligible AS INT),
    TRY_CAST(EUEligibleDuration AS INT),
    ExtraNote,
    CAST(FlightFound AS INT),
    TRY_CAST(LegNo AS INT),
    CAST(IsTimeLimitL1 AS INT),
    CAST(IsTimeLimitL2 AS INT),
    TRY_CAST(EUFlights_Id AS BIGINT),
    Link_Id,
    TRY_CAST(DelayInSecond AS INT) AS DelayInSecond,   -- TRY_CAST handles empty strings gracefully
    Status,
    CAST(IsSingleFlight AS INT)
FROM #staging;
